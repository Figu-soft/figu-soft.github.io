loading.gd 
extends Control

@onready var progress_bar = $ProgressBar

var loading_finished := false

func _ready():
	MobileAds.initialize()
	await start_loading()
	get_tree().change_scene_to_file("res://scenes/main.tscn")


func start_loading():
	while progress_bar.value < 100:
		await get_tree().create_timer(0.05).timeout
		progress_bar.value += 3
main.gd 
extends Node

@onready var interstitial = preload("res://scripts/interstitial.gd").new()
@onready var banner = preload("res://scripts/banner.gd").new()

var player_is_dead := false


func _ready():

	add_child(interstitial)
	add_child(banner)

	interstitial.setup_interstitial()
	interstitial.load_interstitial()

	banner.load_banner()

	$Destroy.pressed.connect(_on_destroy_pressed)
	$Restart.pressed.connect(_on_restart_pressed)


func _on_destroy_pressed():

	if player_is_dead:
		return
	
	var player = $World/Player
	if player:
		player_is_dead = true
		player.queue_free()
		print("Player destroyed")

		await get_tree().create_timer(0.6).timeout
		interstitial.show_interstitial()


func _on_restart_pressed():
	get_tree().reload_current_scene()
banner.gd 
extends Node

var _banner_view: AdView

const TEST_BANNER_ANDROID := "ca-app-pub-3940256099942544/6300978111"


func load_banner():

	if OS.get_name() != "Android":
		return

	if _banner_view:
		return

	_banner_view = AdView.new(
		TEST_BANNER_ANDROID,
		AdSize.BANNER,
		AdPosition.Values.BOTTOM
	)

	var ad_listener := AdListener.new()

	ad_listener.on_ad_failed_to_load = func(load_ad_error : LoadAdError) -> void:
		print("Banner falló: ", load_ad_error.message)

	ad_listener.on_ad_loaded = func() -> void:
		print("Banner cargado")

	ad_listener.on_ad_clicked = func() -> void:
		print("Banner clickeado")

	ad_listener.on_ad_opened = func() -> void:
		print("Banner abierto")

	ad_listener.on_ad_closed = func() -> void:
		print("Banner cerrado")

	ad_listener.on_ad_impression = func() -> void:
		print("Banner impresión registrada")

	_banner_view.ad_listener = ad_listener
	_banner_view.load_ad(AdRequest.new())


func destroy_banner():
	if _banner_view:
		_banner_view.destroy()
		_banner_view = null
interstitial.gd 
extends Node

var _interstitial_ad: InterstitialAd
var _full_screen_content_callback := FullScreenContentCallback.new()

const TEST_INTERSTITIAL_ANDROID := "ca-app-pub-3940256099942544/1033173712"


func setup_interstitial():

	_full_screen_content_callback.on_ad_dismissed_full_screen_content = func() -> void:
		if _interstitial_ad:
			_interstitial_ad.destroy()
			_interstitial_ad = null
			load_interstitial()
		print("Ad cerrado")

	_full_screen_content_callback.on_ad_failed_to_show_full_screen_content = func(_ad_error : AdError) -> void:
		if _interstitial_ad:
			_interstitial_ad.destroy()
			_interstitial_ad = null
			load_interstitial()

	print("Interstitial configurado")


func load_interstitial():

	if _interstitial_ad:
		_interstitial_ad.destroy()
		_interstitial_ad = null

	if OS.get_name() != "Android":
		return

	var interstitial_ad_load_callback := InterstitialAdLoadCallback.new()

	interstitial_ad_load_callback.on_ad_failed_to_load = func(ad_error : LoadAdError) -> void:
		print("Error cargando interstitial: ", ad_error.message)

	interstitial_ad_load_callback.on_ad_loaded = func(interstitial_ad : InterstitialAd) -> void:
		print("Interstitial cargado")
		_interstitial_ad = interstitial_ad
		_interstitial_ad.full_screen_content_callback = _full_screen_content_callback

	InterstitialAdLoader.new().load(
		TEST_INTERSTITIAL_ANDROID,
		AdRequest.new(),
		interstitial_ad_load_callback
	)


func show_interstitial():
	if _interstitial_ad:
		_interstitial_ad.show()
	else:
		print("Interstitial no listo")
